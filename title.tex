%% example.tex
%% Shows how to use the UGent 2016 styles.

% The ugentreport2016 and ugentbook2016 are based on respectively scrreprt and scrbook respectively.
% They have three options:
% - campus: default, global, kortrijk - determines which UGent logo is used
% - faculty: two letter code for the facculty. See the docs for the possible values. Changes logos and colours
% - layout: none, margins, colours, titlestyle, titlefont, all. Possible values are listed below, but are described
%           more completely in the manual. Each next level also applies everything from the previous one.
%   - none: don't apply any styling apart from the title page
%   - margins: adjust the margins of the pages
%   - colours: apply ugent colours to titles
%   - titlestyle: apply styling to titles (bold, font size, etc.)
%   - titlefont: apply the UGent Panno font to the titles (you must have this font installed, see the manual)
%   - all: apply all styling, including the body text
%
% All other options are passed to the underlying class.
\documentclass[11pt,dutch]{scrbook}

\usepackage[ugentstyle=course]{ugentstyle2016}

% Some meta data. See the manual for an overview for everything.
\author{author}
\titletext{titletext}
\title{title}
\subtitle{subtitle}
\academyyear{academyyear}
\programme{programme}
\wordcount{wordcount}
\studentnumber{studentnumber}
\email{email}
\phone{phone}
\address{address}
\department{department}
\researchgroup{researchgroup}
\facultish{facultish}

\begin{document}
    \maketitle
\end{document}
